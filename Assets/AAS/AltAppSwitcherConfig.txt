// AltAppSwitcher config file

// Possible key bindings values:
//     left alt
//     right alt
//     alt
//     tilde
//     left super (left windows)
//     right super (right windows)
//     left control
//     left shift
//     right shift
//     tab
//     none

app hold key: left alt
next app key: tab
previous app key: tilde
window hold key: left alt
next window key: tilde
invert order key: left shift

// Graphic options:
// Theme can be "auto" (matches windows theme), "light" or "dark".
// Scale controls icon size, expressed as a ratio, 1 being Windows default icon size.
theme: auto
scale: 2.5

// Other options:
allow mouse: true
check for updates: true

// App switcher mode:
// "app": MacOS-like, one entry per application.
// "window": Windows-like, one entry per window (each window is considered an independent application)
app switcher mode: app

// Display names under app entries:
// selected: if selected
// all: display all names
// none: do not display any name
display name: selected

// Multiple monitor mode:
// Controls where the switcher UI appears (placement), not which apps are listed.
// "mouse": Place the UI on the monitor where the mouse cursor is located
// "main":  Place the UI on the primary (main) monitor
multiple monitor mode: mouse

// App filter mode:
// Controls which apps/windows are listed.
// "all": Show apps from all monitors
// "mouse monitor": Show only apps from the monitor where the mouse cursor is located
app filter mode: all
